story_fremen_survival = {
    on_setup = {
        story_owner = {
            set_variable = {
                name = fremen_survival_days
                value = 0
            }
        }
    }

    on_monthly = {
        story_owner = {
            change_variable = {
                name = fremen_survival_days
                add = 30
            }

            # After 2 years (730 days), flag as survived
            if = {
                limit = {
                    var:fremen_survival_days >= 730
                    NOT = { has_character_flag = survived_with_fremen }
                }
                add_character_flag = survived_with_fremen
                trigger_event = { id = arrakis_struggle.210 }
            }
        }
    }
